FIND_PACKAGE (Qt4)

INCLUDE_DIRECTORIES (. .. ${QT_INCLUDES} ${CMAKE_CURRENT_BINARY_DIR})

QT4_WRAP_CPP (kephalapp_SRCS kephalapp.h)

TARGET_LINK_LIBRARIES (kephalapp ${QT_LIBRARIES} kephal)
